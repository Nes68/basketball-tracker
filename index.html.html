const CACHE_NAME = "macy-miranda-tracker-v1"; const URLS_TO_CACHE = [ "/basketball-tracker/", "/basketball-tracker/index.html", "/basketball-tracker/manifest.json" ]; self.addEventListener("install", (event) => { event.waitUntil( caches.open(CACHE_NAME).then((cache) => cache.addAll(URLS_TO_CACHE)) ); }); self.addEventListener("activate", (event) => { event.waitUntil( caches.keys().then((names) => Promise.all( names.map((name) => { if (name !== CACHE_NAME) { return caches.delete(name); } }) ) ) ); }); self.addEventListener("fetch", (event) => { event.respondWith( caches.match(event.request).then((response) => { return response || fetch(event.request); }) ); });
